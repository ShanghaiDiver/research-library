let activeFilter="all";

let sortKey="gap", sortDir=-1; // default: most undervalued first

const rowsEl=document.getElementById('rows');

const emptyEl=document.getElementById('empty');

const qEl=document.getElementById('q');


function gap(c){return Math.round((c.fairValue-c.price)/c.price*100);}

function money(cur,v){const n=v.toLocaleString('en-GB');return cur==='$'?'$'+n:n+'p';}

function range(c){const lo=c.buyBelow.toLocaleString('en-GB'),hi=c.trimAbove.toLocaleString('en-GB');

  return c.currency==='$'?`$${lo}–${hi}`:`${lo}–${hi}p`;}

function zone(c){if(c.price<=c.buyBelow)return'buy';if(c.price>=c.trimAbove)return'trim';return'hold';}


const acc={

  name:c=>c.name,

  verdict:c=>({Buy:0,Watch:1,Avoid:2})[c.verdict],

  price:c=>c.price,

  fairValue:c=>c.fairValue,

  gap:c=>gap(c),

  buyBelow:c=>c.buyBelow,

  trimAbove:c=>c.trimAbove

};


function render(){

  const q=qEl.value.trim().toLowerCase();

  let list=companies.filter(c=>{

    const matchF=activeFilter==='all'||c.verdict===activeFilter;

    const matchQ=!q||[c.name,c.ticker,c.sector].join(' ').toLowerCase().includes(q);

    return matchF&&matchQ;

  });

  const get=acc[sortKey];

  list.sort((a,b)=>{

    const av=get(a),bv=get(b);

    if(typeof av==='string')return av.localeCompare(bv)*sortDir;

    return (av-bv)*sortDir;

  });


  rowsEl.innerHTML=list.map(c=>{

    const g=gap(c);

    const gcls=g<0?'neg':'pos';

    const gtxt=(g>0?'+':'')+g+'%';

    const z=zone(c);

    const star=c.prov?'<span class="prov">*</span>':'';

    return `<tr onclick="window.open('${c.file}','_blank')">

      <td><div class="co">${c.name}</div><div class="tk">${c.ticker} · ${c.exchange}</div></td>

      <td><span class="sec">${c.sector}</span></td>

      <td><span class="pill ${c.verdict.toLowerCase()}">${c.verdict}</span></td>

      <td>${money(c.currency,c.price)}</td>

      <td>${money(c.currency,c.fairValue)}</td>

      <td class="gap ${gcls}">${gtxt}</td>

      <td class="bbuy ${z==='buy'?'zone-buy':''}">${money(c.currency,c.buyBelow)}${star}</td>

      <td class="bhold">${range(c)}</td>

      <td class="btrim ${z==='trim'?'zone-trim':''}">${money(c.currency,c.trimAbove)}</td>

      <td><a class="open" href="${c.file}" target="_blank" onclick="event.stopPropagation()">open →</a></td>

    </tr>`;

  }).join('');

  emptyEl.style.display=list.length?'none':'block';

  arrows();

}


function arrows(){

  document.querySelectorAll('th.sortable').forEach(th=>{

    const a=th.querySelector('.arr');

    const on=th.dataset.key===sortKey;

    th.classList.toggle('active',on);

    a.textContent=on?(sortDir===1?' ↑':' ↓'):'';

  });

}


function summary(){

  const n=companies.length;

  const by=v=>companies.filter(c=>c.verdict===v).length;

  const inBuy=companies.filter(c=>zone(c)==='buy').length;

  document.getElementById('summary').innerHTML=

    `<b>${n}</b> companies · <b>${by('Buy')}</b> buy · <b>${by('Watch')}</b> watch · <b>${by('Avoid')}</b> avoid · <b>${inBuy}</b> in buy zone`;

}


document.getElementById('filter').addEventListener('click',e=>{

  if(e.target.tagName!=='BUTTON')return;

  activeFilter=e.target.dataset.v;

  [...e.currentTarget.children].forEach(b=>b.classList.toggle('on',b===e.target));

  render();

});

qEl.addEventListener('input',render);

document.querySelectorAll('th.sortable').forEach(th=>{

  th.addEventListener('click',()=>{

    const k=th.dataset.key;

    if(k===sortKey){sortDir=-sortDir;}

    else{sortKey=k;sortDir=(k==='name')?1:(k==='gap'?-1:1);}

    render();

  });

});


summary();render();
